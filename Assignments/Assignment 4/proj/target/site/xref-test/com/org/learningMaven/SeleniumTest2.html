182      public static void takeSnapShot(WebDriver webdriver,String fileWithPath) throws Exception{
183     	//Convert web driver object to TakeScreenshot
184     	TakesScreenshot scrShot =((TakesScreenshot)webdriver);
185     	//Call getScreenshotAs method to create image file
186     	File SrcFile=scrShot.getScreenshotAs(OutputType.FILE);
187     	//Move image file to new destination
